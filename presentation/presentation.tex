\section{Zielstellung}
\frame{\frametitle{Gliederung} \tableofcontents[currentsection]}

\begin{frame}{Zielstellung}
\begin{itemize}
\item Bienen können sich mit einer Krankheit infizieren die über kurz oder lang zum Zusammenbruch der Population führt
\item Es soll der Einfluss der Bienenstockkonfiguration auf die Ausbreitungsgeschwindigkeit der Krankheit innerhalb der Bienenpopulation untersucht werden
\end{itemize}

\end{frame}

\section{Problemabgrenzung, Daten und Annahmen}

\subsection{Problemabgrenzung}
\frame{\frametitle{Gliederung} \tableofcontents[currentsection]}

\begin{frame}
\frametitle{Problemabgrenzung (1)}
\begin{itemize}
\item Imker besitzt bestimmte Anzahl Bienenstöcke die auf einer fest definierten Fläche platziert werden können
\item Königin produziert solange Bienen bis Kapazität des Stocks erreicht ist
\item Bienen fliegen durch den freien Raum zu Blüten, sammeln Nektar, bringen diesen zurück und ruhen sich aus
\item Bienen sterben entweder nach einer gewissen Zeit oder wenn sie sich mit der Krankheit infiziert haben
\item Erkrankte Bienen können auch vor Ablauf der Inkubationszeit andere Bienen anstecken
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Problemabgrenzung (2)}
\begin{itemize}
\item Eine Ansteckung ist entweder durch das gemeinsame Sammeln an einer Blüte oder der gemeinsame Aufenthalt im Stock möglich
\item Sobald Biene Symptome zeigt, wird sie vom Stock ausgestoßen, fliegt ziellos umher und stirbt nach kurzer Zeit ohne weitere Bienen anzustecken % Annahme von uns??
\item Falls sich Bienen zu einem fremden Stock verirren, dürfen sie ihren Nektar abladen, werden dann aber von dessen Wachen fortgeschickt.\\
$\Rightarrow$ Kann ebenfalls zu einer Ansteckung führen

\end{itemize}
\end{frame}

\subsection{Annahmen}
\frame{\frametitle{Gliederung} \tableofcontents[currentsection]}

\begin{frame}
\frametitle{Annahmen}
\begin{itemize}
\item In Natur werden verschiedenen Bienen verschiedene Aufgaben zugeteilt\\
$\Rightarrow$ Junge Bienen umsorgen den Stock und die Eier während Ältere Bienen Nektar sammeln
\item Im Model wird auf diese Teilung verzichtet und davon ausgegangen das immer ein fester Prozentsatz an Bienen sich im Stock befindet
\item Es wird davon ausgegangen das initial ein bestimmter Anteil Bienen erkrankt ist
\end{itemize}
\end{frame}

\subsection{Eingabeaten}
\frame{\frametitle{Gliederung} \tableofcontents[currentsection]}

\begin{frame}
\frametitle{Eingabedaten}
\begin{itemize}
\item Verfügbare Imkerfläche und maximale Anzahl Bienenstöcke des Imkers
\item Lebensdauer, Tragekapazität und Fluggeschwindigkeit der Biene
\item Nektarmenge und Regenerationsdauer einer Blüte
\item Verweildauer einer Biene im eigenen/fremden Stock
\item Wahrscheinlichkeit für Verfliegen zum fremden Stock
\item Eierlegerate einer Königin und Kapazität eines Bienenstocks
\item Anteil der Sammler im Stock
\item Anteil initial erkrankter Bienen, Inkubationszeit und Ansteckungswahrscheinlichkeit für andere Bienen
\end{itemize}
\end{frame}

\section{Einleitung}
\frame{\frametitle{Gliederung} \tableofcontents[currentsection]}

\begin{frame}{Problem- und Zielstellung}
\textbf{Problemstellung}
\begin{itemize}
\item Bienen können sich mit einer Krankheit infizieren
\item erkrankte Bienen können kompletten Bienenstock anstecken
\item dies kann zum Absterben des Stocks führen
\end{itemize}

\vspace*{1em}

\textbf{Zielstellung}
\begin{itemize}
\item Einfluss der Bienenstockplatzierung auf Ausbreitungsgeschwindigkeit der Krankheit innerhalb der Bienenpopulation untersuchen
\end{itemize}

\end{frame}

\section{Simulationsmodell}
\frame{\frametitle{Gliederung} \tableofcontents[currentsection]}

\begin{frame}
\frametitle{Problemabgrenzung}
\textbf{Situation}
\begin{itemize}
\item Imker besitzen eine bestimmte Anzahl Bienenstöcke, die er auf einer festgelegten Fläche platzieren kann
\end{itemize}

\vspace*{1em}
\textbf{Leben einer Biene}
\begin{itemize}
\item eine Bienenkönigin legt solange Eier bis Kapazität ihres Stocks erreicht ist
\item Bienen fliegen zu Blüten, sammeln Nektar, bringen diesen zurück und ruhen sich aus
\item Bienen können versehentlich zu einem falschen Stock fliegen
	\begin{itemize}
	\item[$\Rightarrow$] nachdem sie dort ihren Nektar abgeliefert haben, werden sie aus dem Stock entfernt
	\end{itemize}
\end{itemize}

\end{frame}

\begin{frame}
\frametitle{Problemabgrenzung}
\textbf{Krankheit und Tod}
\begin{itemize}
\item Bienen sterben entweder eines natürlichen Todes oder durch die Krankheit
\item erkrankte Bienen können andere Bienen anstecken
\item eine erkrankte Biene kann weitere Bienen im Stock oder an Blüten anstecken
\item sobald eine Biene Symptome zeigt, wird sie vom Stock ausgestoßen % fliegt ziellos umher und stirbt nach kurzer Zeit ohne weitere Bienen anzustecken % Annahme von uns??
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Annahmen}
\textbf{Aufgabenverteilung}
\begin{itemize}
\item in der Natur werden den Bienen je nach Alter verschiedene Aufgaben zugeteilt (Bienenstock umsorgen, Nektar sammeln)
\begin{itemize}
\item[$\Rightarrow$] 
 im Modell wird darauf verzichtet: Es gibt Sammelbienen und Stockbienen
\end{itemize}

\vspace*{1em}
\end{itemize}
\textbf{Initialer Krankheitszustand}
\begin{itemize}
\item ein bestimmter Prozentsatz Bienen ist am Anfang der Simulation bereits erkrankt
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Eingabedaten}
\textbf{Imker}
\begin{itemize}
\item verfügbare Anbaufläche
\item Anzahl Bienenstöcke
\end{itemize}

\vspace*{0.5em}
\textbf{Bienen und Blüten}
\begin{itemize}
\item Eierlegerate einer Königin
\item Kapazität eines Bienenstocks
\item Anteil der Sammler im Stock
\item Lebensdauer einer Biene
\item Nektartragekapazität einer Biene
\item Fluggeschwindigkeit einer Biene
\item $\ldots$
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Eingabedaten}
\begin{itemize}
\item $\ldots$
\item Verweildauer einer Biene im Stock (nach Nektarlieferung)
\item Wahrscheinlichkeit für Verfliegen zu fremdem Stock
\item Nektarmenge und Regenerationsdauer einer Blüte
\end{itemize}

\vspace*{1em}
\textbf{Krankheit}
\begin{itemize}
\item Anteil initial erkrankter Bienen
\item Inkubationszeit und Ansteckungswahrscheinlichkeit für gesunde Bienen
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Ziel- und Stellgröße}
\textbf{Zielgröße}
\begin{itemize}
\item Zeitdauer bis komplette Bienenpopulation kollabiert
\begin{itemize}
\item[$\Rightarrow$]  Maximierungsproblem
\end{itemize}
\end{itemize}
\vspace*{1em}
\textbf{Stellgröße}
\begin{itemize}
\item Positionierung der Bienenstöcke auf der verfügbaren Fläche
\end{itemize}
\end{frame}

\section{Schlussbetrachtung}
\frame{\frametitle{Gliederung} \tableofcontents[currentsection]}
\begin{frame}
\frametitle{Schlussbetrachtung}
\textbf{Zusammenfassung}
\begin{itemize}
	\item keine genaue Aussage bezüglich optimaler Konfiguration möglich
	\item viele kleine Gruppen verringern offensichtlich die Varianz
\end{itemize}

\vspace*{1em}
\textbf{Kritische Würdigung}
\begin{itemize}
\item genetische Diversität zwischen Bienenvölkern unbetrachtet
\item Bienen können Immunität besitzen oder entwickeln
\item Imker kann Medikamente verabreichen
\item nicht alle möglichen Bienenstockplatzierungen betrachtet
\item zu wenig Wiederholungen, Bienen oder Stöcke
\end{itemize}

\section{Zielstellung}
\frame{\frametitle{Gliederung} \tableofcontents[currentsection]}

\begin{frame}{Zielstellung}
\textbf{Problemstellung}
\begin{itemize}
\item Bienen können sich mit einer Krankheit infizieren
\item erkrankte Bienen können Krankheit kompletten Bienenstock anstecken
\item dies kann zum Absterben eines Stocks führen
\end{itemize}

\vspace*{1em}

\textbf{Zielstellung}
\begin{itemize}
\item daher soll Einfluss der Bienenstockplatzierung auf Ausbreitungsgeschwindigkeit der Krankheit innerhalb der Bienenpopulation untersucht werden
\end{itemize}

\end{frame}

\section{Problemabgrenzung, Daten und Annahmen}

\subsection{Problemabgrenzung}
\frame{\frametitle{Gliederung} \tableofcontents[currentsubsection]}

\begin{frame}
\frametitle{Problemabgrenzung (1)}
\begin{itemize}
\item Imker besitzen eine bestimmte Anzahl Bienenstöcke, die auf einer festgelegten Fläche platziert werden sollen
\item eine Bienenkönigin legt solange Eier bis Kapazität ihres Stocks erreicht ist
\item Bienen fliegen zu Blüten, sammeln Nektar, bringen diesen zurück und ruhen sich aus
\item Bienen sterben entweder eines natürlichen Todes oder durch die Krankheit
\item erkrankte Bienen können vor Ablauf der Inkubationszeit andere Bienen anstecken
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Problemabgrenzung (2)}
\begin{itemize}
\item eine erkrankte Biene kann weitere Bienen im Stock oder an Blüten anstecken
\item sobald eine Biene Symptome zeigt, wird sie vom Stock ausgestoßen, fliegt ziellos umher und stirbt nach kurzer Zeit ohne weitere Bienen anzustecken % Annahme von uns??
\item Bienen können versehentlich zu einem falschen Stock fliegen
	\begin{itemize}
	\item nachdem sie dort ihren Nektar abgegeben haben, werden sie aus dem Stock entfernt
	\item während des Aufenthalts können andere Bienen angesteckt werden
	\end{itemize}
\end{itemize}
\end{frame}

\subsection{Annahmen}
\frame{\frametitle{Gliederung} \tableofcontents[currentsubsection]}

\begin{frame}
\frametitle{Annahmen}
\begin{itemize}
\item in Natur werden verschiedenen Bienen, verschiedene Aufgaben zugeteilt (Bienenstock umsorgen, Nektar sammeln)
\begin{itemize}
\item[$\Rightarrow$] 
 im Modell wird darauf verzichtet und es befindet sich immer ein bestimmter Anteil im Stock
\end{itemize}
\item ein bestimmter Prozentsatz Bienen ist am Anfang der Simulation bereits erkrankt
\end{itemize}
\bigskip
\footnotesize\cite[Vgl.][]{tautz2007phaenomen} \\
\footnotesize\cite[Vgl.][]{genersch2010honey} \\
\footnotesize\cite[Vgl.][]{tautz2013bee} \\
\end{frame}

\subsection{Eingabedaten}
\frame{\frametitle{Gliederung} \tableofcontents[currentsubsection]}

\begin{frame}
\frametitle{Eingabedaten}
\begin{itemize}
\item verfügbare Anbaufläche und maximale Anzahl Bienenstöcke des Imkers
\item Lebensdauer, Tragekapazität, Fluggeschwindigkeit einer Biene
\item Nektarmenge und Regenerationsdauer einer Blüte
\item Verweildauer einer Biene im eigenen/fremden Stock
\item Wahrscheinlichkeit für Verfliegen zu fremdem Stock
\item Eierlegerate einer Königin und Kapazität eines Bienenstocks
\item Anteil der Sammler im Stock
\item Anteil initial erkrankter Bienen
\item Inkubationszeit und Ansteckungswahrscheinlichkeit für gesunde Bienen
\end{itemize}
